GitHub Actions  --> Workflow (.GitHub/workflow)
==============

ci/cd.yml
------------

name: CI/CD pipeline

on: 
  push:
    branches: [main]

    path-ignore:
     -Kubernetes/deployment.yml

  pull_request:
    branch: [main]

jobs:

  test:
    name: Unit testing
    runs-on: ubuntu-latest
    
    steps:
      - name: checkout code
        uses: actions/checkout@v4

      - name: setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        uses: npm ci
   
      - name: Run tests
        uses: npm test || echo "No tests found,would add tests in a real project"
 
  lint:
    name: Static code Analysis
    runs-on: ubuntu-latest
    
    steps:
      - name: checkout code
        uses: actions/checkout@v4

      - name: setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        uses: npm ci
   
      - name: Run ESLint
        uses: npm run lint

  build:
    name: Build
    runs-on: ubuntu-latest
    needs: [test,lint]
    steps:
     
      - name: checkout code
        uses: actions/checkout@v4

      - name: setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        uses: npm ci
   
      - name: RUN build
        uses: npm run build
   
      - name: Upload build artifacts
        uses: actions/upload-artifact@v4
        with:
          name: build-artifacts
          path: dist/
